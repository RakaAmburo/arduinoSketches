│    - Memoria suficiente (no usar SPIFFS con OTA)    │
├─────────────────────────────────────────────────────┤
│ 5. TROUBLESHOOTING:                                 │
│    - No aparece IP: Revisar WiFi/Monitor Serial     │
│    - Fallo upload: Reiniciar ESP, revisar red       │
│    - Lento: Red WiFi estable, cerca del router      │
└─────────────────────────────────────────────────────┘

#include <ArduinoOTA.h>
void setup() {
  Serial.begin(115200);
  WiFi.begin("SSID", "PASS");
  while (WiFi.status() != WL_CONNECTED) delay(500);
  Serial.print("IP: "); Serial.println(WiFi.localIP());
  
  ArduinoOTA.setPassword("1234");
  ArduinoOTA.begin();
}
void loop() { ArduinoOTA.handle(); }
